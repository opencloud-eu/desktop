cmake_minimum_required(VERSION 3.18)
project(FontIconCollector)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Qt6 REQUIRED COMPONENTS Core Widgets Gui)

qt6_standard_project_setup()

add_executable(fonticon_collector
    fonticon_collector_simple.cpp
)

target_link_libraries(fonticon_collector 
    Qt6::Core 
    Qt6::Widgets 
    Qt6::Gui
)

# Enable Qt's MOC compiler for the Q_OBJECT macro
set_target_properties(fonticon_collector PROPERTIES
    AUTOMOC ON
)
